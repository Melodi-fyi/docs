---
title: 'Get Threads'
description: 'Retrieve threads with filtering options for pagination, IDs, users, search terms, projects, intents, segments, and date ranges'
api: GET https://app.melodi.fyi/api/external/threads
---

## Parameters
<ParamField query="pageSize" type="integer" default="50">
  Number of threads to return per page.
</ParamField>

<ParamField query="pageIndex" type="integer" default="0">
  Page index for pagination.
</ParamField>

<ParamField query="ids" type="string">
  List of thread IDs to filter by (comma-separated integers).
</ParamField>

<ParamField query="externalUserIds" type="string">
  List of external user IDs to filter by (comma-separated integers).
</ParamField>

<ParamField query="search" type="string">
  Search term to filter threads.
</ParamField>

<ParamField query="userSearch" type="string">
  Search term to filter by user.
</ParamField>

<ParamField query="projectId" type="integer">
  ID of the project to filter threads by.
</ParamField>

<ParamField query="intentIds" type="string">
  List of intent IDs to filter by (comma-separated integers).
</ParamField>

<ParamField query="userSegmentIds" type="string">
  List of user segment IDs to filter by (comma-separated integers).
</ParamField>

<ParamField query="before" type="string">
  Filter threads created before this date (ISO 8601 format, e.g. 2024-03-21T00:00:00Z).
</ParamField>

<ParamField query="after" type="string">
  Filter threads created after this date (ISO 8601 format, e.g. 2024-03-21T00:00:00Z).
</ParamField>

## Response fields

<ResponseField name="count" type="integer" required>
  Total number of threads.
</ResponseField>

<ResponseField name="rows" type="array" required>
  Array of thread objects.
</ResponseField>

<ResponseField name="rows[].id" type="number">
  Unique identifier for the thread.
</ResponseField>

<ResponseField name="rows[].externalId" type="string">
  External thread ID.
</ResponseField>

<ResponseField name="rows[].organizationId" type="number">
  ID of the organization to which the thread belongs.
</ResponseField>

<ResponseField name="rows[].projectId" type="number">
  ID of the project to which the thread belongs.
</ResponseField>

<ResponseField name="rows[].metadata" type="object">
  Metadata associated with the thread.
</ResponseField>

<ResponseField name="rows[].createdAt" type="string" format="date-time">
  Date when the thread was created.
</ResponseField>

<ResponseField name="rows[].updatedAt" type="string" format="date-time">
  Date when the thread was last updated.
</ResponseField>
